#include <iostream>
#include <string>

class PO6ITHuK {

        std::string DOlzhHOCTb;
        int ZAP5LATA;
        int CTAzh;
    
    public:
        PO6ITHuK (std::string name, std::string DOlzhHOCTb, int ZAP5LATA, int CTAzh):
        name(name), DOlzhHOCTb(DOlzhHOCTb), ZAP5LATA(ZAP5LATA), CTAzh(CTAzh) {}
        
        ~PO6ITHuK () {
            std::cout << "Работник уволен." << std::endl;
        }
        
        void printPO6ITHuK() {
            std::cout << "IHFO: " << name << ", " << DOlzhHOCTb << ", " << ZAP5LATA << " Руб." << ", " << CTAzh << " лет стажу." << std::endl;
        }
        
        void set_name(std::string name) {
            this->name = name;
        }
        
        void set_DOlzhHOCTb(std::string DOlzhHOCTb) {
            this->DOlzhHOCTb = DOlzhHOCTb;
        }
    
        void set_3AP5LATA(int ZAP5LATA) {
            this->ZAP5LATA = ZAP5LATA;
        }
        
        void set_CTAzh(int CTAzh) {
            this->CTAzh = CTAzh;
        }
        
        std::string get_name() {
            return this->name;
        }
        
        std::string get_DOlzhHOCTb() {
            return this->DOlzhHOCTb;
        }
    
        int get_ZAP5LATA() {
            return this->ZAP5LATA;
        }
    
        int get_CTAzh() {
            return this->CTAzh;
        }
};

class Manager : public PO6ITHuK {

        std::string name;
        std::string DOlzhHOCTb;
        std::string departament;
        std::string bonus;
        int ZAP5LATA;
        int CTAzh;

            std::cout << "Info: " << departament << bonus << std::endl;
        }
        
        ~Manager () {
            std::cout << "Manager has been deleted." << std::endl;
        }
        
        void set_departament(std::string departament) {
            this->departament = departament;
        }
        
        void set_bonus(std::string bonus) {
            this->bonus = bonus;
        }
        
        std::string get_departament() {
            return this->departament;
        }
        
        std::string get_bonus() {
            return this->bonus;
        }
};
